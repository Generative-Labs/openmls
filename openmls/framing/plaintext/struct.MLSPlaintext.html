MLSPlaintext is a framing structure for MLS messages. It can contain Proposals, Commits and application messages.

  1. Message framing
struct {
    opaque group_id<0..255>;
    uint64 epoch;
    Sender sender;
    opaque authenticated_data<0..2^32-1>;

    ContentType content_type;
    select (MLSPlaintext.content_type) {
        case application:
            opaque application_data<0..2^32-1>;

        case proposal:
            Proposal proposal;

        case commit:
            Commit commit;
    }

opaque signature<0..2^16-1>;
optional<MAC> confirmation_tag;
optional<MAC> membership_tag;

} MLSPlaintext;
